```sql
create table emp (
	ssn char(9) primary key
) engine=InnoDB;
```

```sql
create table dept(
	dname char(20) not null,
    dnumber integer not null,
    mgrssn char(9),
    mgrstartdate date,
    primary key (dnumber),
    unique(dname),
    foreign key (mgrssn) references emp(ssn)
    on delete cascade
) ENGINE=InnoDB;
```
